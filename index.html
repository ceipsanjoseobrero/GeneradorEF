window.generar = async function(){

- Participación activa

- Toma de decisiones

Calentamiento + juego + vuelta a la calma

});

// 📅 CALENDARIO

async function cargarCalendario(){

grid.innerHTML="";

for(let i=1;i<=30;i++){

const d=document.createElement("div");

d.className="day";

d.textContent=i;


d.onclick=async()=>{

const snap = await getDocs(collection(db,"sesiones"));

let txt="";

snap.forEach(doc=>{

if(doc.data().dia===i) txt+=doc.data().texto+"\n\n";

});

detalle.textContent=txt||"Sin sesiones";

};


grid.appendChild(d);

}

}


// ⏰ HORARIO TIEMPO REAL

function escucharHorario(){

onValue(ref(rtdb,'horario'), snap=>{

const data = snap.val() || {horas:["8:30"], cursos:["1º ESO"]};


let html="<table>";

data.horas.forEach(h=>{

html+="<tr><td>"+h+"</td>";

data.cursos.forEach(c=>{

html+=`<td contenteditable="true" onblur="guardarCambio('${h}',this.innerText)">${c}</td>`;

});

html+="</tr>";

});

html+="</table>";


tabla.innerHTML=html;

});

}


window.guardarCambio = function(hora,valor){

onValue(ref(rtdb,'horario'), snap=>{

let data = snap.val();

if(!data) return;

data.cursos[0]=valor;

set(ref(rtdb,'horario'), data);

},{onlyOnce:true});

}


window.addCursoRT = function(){

onValue(ref(rtdb,'horario'), snap=>{

let data = snap.val()||{horas:[],cursos:[]};

data.cursos.push(nuevoCurso.value);

set(ref(rtdb,'horario'), data);

},{onlyOnce:true});

}


window.addHoraRT = function(){

onValue(ref(rtdb,'horario'), snap=>{

let data = snap.val()||{horas:[],cursos:[]};

data.horas.push(nuevaHora.value);

set(ref(rtdb,'horario'), data);

},{onlyOnce:true});

}

const juegosBase=[];

for(let i=1;i<=300;i++){

juegosBase.push({nombre:"Juego "+i,tipo:categorias[i%categorias.length]});

}


async function cargarJuegosIniciales(){

const snap=await getDocs(collection(db,"juegos"));

if(snap.empty){

for(const j of juegosBase){

await addDoc(collection(db,"juegos"), j);

}

}

cargarJuegos();

}


async function cargarJuegos(){

listaJuegos.innerHTML="";

const snap=await getDocs(collection(db,"juegos"));

snap.forEach(doc=>{

const j=doc.data();

listaJuegos.innerHTML+=`<div class="juegoItem">${j.nombre} (${j.tipo})</div>`;

});

}


window.filtrarJuegos = function(){

const f=filtroJuego.value.toLowerCase();

document.querySelectorAll(".juegoItem").forEach(el=>{

el.style.display = el.textContent.toLowerCase().includes(f) ? "block":"none";

});

}
